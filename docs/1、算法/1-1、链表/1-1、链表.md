## 2. 算法题
给你一个链表，每 k 个节点一组进行翻转，请你返回翻转后的链表(leetcode25)
```
public ListNode reverseKGroup(ListNode head, int k) {
    int i = 1;
    ListNode preTail = null, curHead = head, res = null;
    while (head != null) {
        ListNode next = head.next;
        if (i % k == 0) {
            head.next = null;
            ListNode newHead = reverse(curHead);
            if (preTail != null) {
                preTail.next = newHead;
            } else {
                res = newHead;
            }
            preTail = curHead;
            preTail.next = next;
            curHead = next;
        }
        head = next;
        i++;
    }
    return res;
}

public ListNode reverse(ListNode head) {
    ListNode pre = null;
    while (head != null) {
        ListNode next = head.next;
        head.next = pre;
        pre = head;
        head = next;
    }
    return pre;
}
```
